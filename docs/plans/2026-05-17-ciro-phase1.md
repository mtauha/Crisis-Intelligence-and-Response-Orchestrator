**Goal:** Build the three-agent SequentialAgent pipeline (Sentinel → Analyst → Commander) to detect a critical Urban Flooding incident and write the full incident to the PostgreSQL database without API/Flutter involvement.

**Architecture:** A Blackboard pattern using PostgreSQL as the state mechanism, orchestrated by Vertex AI Agent Engine reasoning. The agents are split into Sentinel (Collector), Analyst (Detector), and Commander (Orchestrator). The pipeline will support local invocation and Gemini stub generation.

**Tech Stack:** Python, PostgreSQL, Alembic, Vertex AI ADK, pytest.

```python
from sqlmodel import Session, select
from ciro.db.models import CityState

def escalate_city_state(session: Session, city: str) -> bool:
    state = session.exec(select(CityState).where(CityState.city == city)).first()
    if state:
        state.status = "critical"
        session.commit()
        return True
    return False

# ciro/agents/commander/actions/routes.py
from sqlmodel import Session, select
from ciro.db.models import Route

def block_flood_routes(session: Session, city: str) -> bool:
    routes = session.exec(select(Route).where(Route.city == city)).all()
    for route in routes:
        route.status = "blocked"
    session.commit()
    return True
```

```python
# ciro/agents/commander/commander_agent.py
from google.adk.agents import BaseAgent
from sqlmodel import Session
from ciro.db.models import Incident
from ciro.agents.commander.actions.city_state import escalate_city_state
from ciro.agents.commander.actions.routes import block_flood_routes
from ciro.agents.commander.actions.tickets import issue_ticket
import uuid

class CommanderAgent(BaseAgent):
    def __init__(self):
        super().__init__()
        self.name = "Commander"

    def execute(self, session: Session, incident_id: uuid.UUID) -> bool:
        incident = session.get(Incident, incident_id)
        if not incident or incident.status not in ["pending_commander", "approved"]:
            return False
            
        # Execute deterministic actions based on incident city
        city = incident.city
        escalate_city_state(session, city)
        block_flood_routes(session, city)
        issue_ticket(session, incident_id, title="Flood response blocked routes", authority="City Traffic")
        
        # Mark actioned
        incident.status = "actioned"
        session.commit()
        return True
```

```python
# ciro/stubs/record_stubs.py
import json
import os
import pathlib

def load_stub(agent_name: str, scenario: str) -> dict:
    stub_path = pathlib.Path(__file__).parent / f"{agent_name.lower()}_{scenario}.json"
    if stub_path.exists():
        with open(stub_path, "r") as f:
            return json.load(f)
    return {}

def save_stub(agent_name: str, scenario: str, data: dict):
    stub_path = pathlib.Path(__file__).parent / f"{agent_name.lower()}_{scenario}.json"
    with open(stub_path, "w") as f:
        json.dump(data, f, indent=2)

# ciro/agents/pipeline.py

def run_pipeline(session: Session, scenario: str) -> bool:
    gemini_enabled = os.environ.get("GEMINI_ENABLED", "true").lower() == "true"

        if analyst_stub:
            incident_id = write_incident(
                session, "urban_flooding", scenario, analyst_stub.get("severity", "critical"), 
                0.95, analyst_stub.get("reasoning", "Stub reason"), {}
            )
            CommanderAgent().execute(session, incident_id)
        return True
    
    # Live execution logic goes here when GEMINI_ENABLED=true
    return True
```
